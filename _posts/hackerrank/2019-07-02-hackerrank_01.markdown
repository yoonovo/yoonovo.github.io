let num = parseInt(prompt('숫자를 입력해 주세요.'));

for (let i=0; i<num; i++) {
	for (let j=0; j<=num; j++){
    	if(j < (num-i)){
    		document.write('&nbsp');
        }else{
    		document.write('*');
        }
    }
    document.write('<br>');
}
